Usage

              
// File: app.component.ts
// other imports...
import {{ '{' }} QRCodeComponent } from 'angularx-qrcode';

@Component({{ '{' }}
// other properties like selector, templateUrl or styleUrls
// selector: "app-root",
imports: [
QRCodeComponent
],
})
export class AppComponent {{
                  '{
  // your code
'
                }} }
